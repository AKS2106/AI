Business Problem

ShopFast Inc. is a mid-size e-commerce company processing ~500K orders/month across web, mobile, and marketplace channels. Their data lives in 4 silos — a transactional MySQL database, a Shopify API, a product catalog in JSON, and customer behavior events from Snowplow. The analytics team cannot answer basic questions like "What is our revenue by category this week?" or "Which customers are at churn risk?"

Why it matters

  • Marketing spends $2M/month with no ROI visibility by channel
  • Warehouse team manually reconciles orders in Excel — 3 days lag
  • Customer support has no unified view of a customer's order history
  • No SLA monitoring on delivery promises

S3 Upload Script (Ingestion)

# src/ingestion/upload_to_s3.py
import boto3, os, logging
from pathlib import Path

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("ingestion")

def upload_to_s3(local_path: str, bucket: str, s3_prefix: str):
    s3 = boto3.client("s3")
    path = Path(local_path)
    success, failed = [], []
    for file in path.rglob("*"):
        if file.is_file():
            key = f"{s3_prefix}/{file.relative_to(path)}"
            try:
                s3.upload_file(str(file), bucket, key)
                success.append(key)
                logger.info(f"✅ Uploaded {key}")
            except Exception as e:
                failed.append({"file": key, "error": str(e)})
                logger.error(f"❌ Failed {key}: {e}")
    return {"uploaded": len(success), "failed": failed}

if __name__ == "__main__":
    result = upload_to_s3(
        local_path="data/raw",
        bucket="shopfast-data-lake",
        s3_prefix="raw"
    )
    print(f"Upload complete: {result}")

Data Quality Checks Module

# src/transformation/data_quality.py
from pyspark.sql import DataFrame
from pyspark.sql.functions import col, count, when
import logging

logger = logging.getLogger("data_quality")

class DataQualityChecker:
    def __init__(self, df: DataFrame, table_name: str):
        self.df = df
        self.table = table_name
        self.results = []

    def check_nulls(self, critical_cols: list) -> "DataQualityChecker":
        for c in critical_cols:
            null_count = self.df.filter(col(c).isNull()).count()
            passed = null_count == 0
            self.results.append({
                "check": f"null_check:{c}", "passed": passed,
                "detail": f"{null_count} nulls"
            })
            if not passed:
                logger.warning(f"⚠️  {self.table}.{c}: {null_count} nulls found")
        return self

    def check_duplicates(self, key_col: str) -> "DataQualityChecker":
        total = self.df.count()
        distinct = self.df.select(key_col).distinct().count()
        dup_count = total - distinct
        passed = dup_count == 0
        self.results.append({"check": f"dup_check:{key_col}",
                              "passed": passed, "detail": f"{dup_count} duplicates"})
        return self

    def check_range(self, col_name: str, min_val, max_val) -> "DataQualityChecker":
        out_of_range = self.df.filter(
            (col(col_name) < min_val) | (col(col_name) > max_val)
        ).count()
        passed = out_of_range == 0
        self.results.append({"check": f"range_check:{col_name}",
                              "passed": passed, "detail": f"{out_of_range} out of range"})
        return self

    def summary(self):
        passed = sum(1 for r in self.results if r["passed"])
        total  = len(self.results)
        logger.info(f"DQ Summary [{self.table}]: {passed}/{total} checks passed")
        if passed < total:
            raise ValueError(f"Data quality failed for {self.table}. Check logs.")
        return self.results

CREATE TABLE dim_customer (
    customer_sk     BIGINT IDENTITY(1,1) PRIMARY KEY,
    customer_id     VARCHAR(20)  NOT NULL,
    full_name       VARCHAR(200),
    email           VARCHAR(200),
    city            VARCHAR(100),
    state           CHAR(2),
    tier            VARCHAR(20),
    effective_date  DATE         NOT NULL,
    expiry_date     DATE         DEFAULT '9999-12-31',
    is_current      BOOLEAN      DEFAULT TRUE,
    created_at      TIMESTAMP    DEFAULT GETDATE()
) DISTSTYLE ALL;  -- small table, replicate to all nodes

CREATE TABLE dim_product (
    product_sk   BIGINT IDENTITY(1,1) PRIMARY KEY,
    product_id   VARCHAR(20)   NOT NULL,
    product_name VARCHAR(300),
    category     VARCHAR(100),
    unit_price   DECIMAL(10,2),
    vendor       VARCHAR(200),
    updated_at   TIMESTAMP
) DISTSTYLE ALL;

CREATE TABLE dim_date (
    date_sk      INT PRIMARY KEY,  -- YYYYMMDD format e.g. 20240115
    full_date    DATE NOT NULL,
    year         SMALLINT,
    month        SMALLINT,
    day          SMALLINT,
    quarter      SMALLINT,
    day_of_week  VARCHAR(10),
    is_weekend   BOOLEAN
) DISTSTYLE ALL;

CREATE TABLE fact_orders (
    order_id       VARCHAR(30)    NOT NULL,
    customer_sk    BIGINT         REFERENCES dim_customer(customer_sk),
    product_sk     BIGINT         REFERENCES dim_product(product_sk),
    date_sk        INT            REFERENCES dim_date(date_sk),
    quantity       SMALLINT,
    unit_price     DECIMAL(10,2),
    discount_pct   DECIMAL(5,2),
    gross_revenue  DECIMAL(12,2),  -- quantity * unit_price
    net_revenue    DECIMAL(12,2),  -- gross * (1 - discount/100)
    order_status   VARCHAR(20),
    channel        VARCHAR(30),
    created_at     TIMESTAMP
)
DISTKEY(customer_sk)       -- co-locate with joins on customer
SORTKEY(date_sk, category_sk);  -- optimize date-range queries

Complex Analytical SQL

1. Revenue by category with MoM growth (Window Function + CTE)

WITH monthly_rev AS (
    SELECT
        p.category,
        d.year,
        d.month,
        SUM(f.net_revenue)    AS revenue,
        COUNT(f.order_id)     AS order_count
    FROM fact_orders f
    JOIN dim_product p ON f.product_sk = p.product_sk
    JOIN dim_date    d ON f.date_sk    = d.date_sk
    WHERE f.order_status != 'CANCELLED'
    GROUP BY 1,2,3
),
growth AS (
    SELECT *,
        LAG(revenue) OVER (
            PARTITION BY category ORDER BY year, month
        ) AS prev_month_rev
    FROM monthly_rev
)
SELECT
    category, year, month, revenue, order_count,
    ROUND((revenue - prev_month_rev) / NULLIF(prev_month_rev,0) * 100, 2) AS mom_growth_pct
FROM growth
ORDER BY category, year, month;

2. Customer RFM segmentation (Ranking + CASE)

WITH rfm_raw AS (
    SELECT
        c.customer_id,
        c.full_name,
        DATEDIFF(day, MAX(d.full_date), CURRENT_DATE) AS recency_days,
        COUNT(DISTINCT f.order_id)                       AS frequency,
        SUM(f.net_revenue)                                AS monetary
    FROM fact_orders f
    JOIN dim_customer c ON f.customer_sk = c.customer_sk
    JOIN dim_date     d ON f.date_sk     = d.date_sk
    WHERE c.is_current = TRUE
    GROUP BY 1,2
),
rfm_scored AS (
    SELECT *,
        NTILE(5) OVER (ORDER BY recency_days  ASC)  AS r_score,
        NTILE(5) OVER (ORDER BY frequency      DESC) AS f_score,
        NTILE(5) OVER (ORDER BY monetary       DESC) AS m_score
    FROM rfm_raw
)
SELECT *,
    CASE
        WHEN r_score >= 4 AND f_score >= 4 THEN 'Champion'
        WHEN r_score >= 3 AND f_score >= 3 THEN 'Loyal Customer'
        WHEN r_score >= 4 AND f_score <= 2 THEN 'New Customer'
        WHEN r_score <= 2 AND f_score >= 3 THEN 'At Risk'
        ELSE 'Needs Attention'
    END AS rfm_segment
FROM rfm_scored
ORDER BY monetary DESC;

3. Running total + top-N per category

WITH ranked AS (
    SELECT
        p.category,
        p.product_name,
        SUM(f.net_revenue)  AS total_revenue,
        RANK() OVER (
            PARTITION BY p.category
            ORDER BY SUM(f.net_revenue) DESC
        ) AS rank_in_category,
        SUM(SUM(f.net_revenue)) OVER (
            PARTITION BY p.category
            ORDER BY SUM(f.net_revenue) DESC
            ROWS UNBOUNDED PRECEDING
        ) AS running_total
    FROM fact_orders f
    JOIN dim_product p ON f.product_sk = p.product_sk
    GROUP BY p.category, p.product_name
)
SELECT * FROM ranked WHERE rank_in_category <= 5;

PySpark Glue ETL Job

# src/transformation/glue_etl_orders.py
import sys, logging
from awsglue.transforms import *
from awsglue.utils import getResolvedOptions
from pyspark.context import SparkContext
from awsglue.context import GlueContext
from pyspark.sql import functions as F
from pyspark.sql.types import *
from data_quality import DataQualityChecker

logger = logging.getLogger("GlueJob")
args   = getResolvedOptions(sys.argv, ["JOB_NAME", "process_date", "env"])
sc     = SparkContext()
glueCtx = GlueContext(sc)
spark  = glueCtx.spark_session

BUCKET   = "shopfast-data-lake"
ENV      = args["env"]  # dev / prod
PROC_DT  = args["process_date"]  # e.g. "2024-01-15"
year, month, day = PROC_DT.split("-")

## ── 1. READ RAW ────────────────────────────────────
orders_schema = StructType([
    StructField("order_id",     StringType(), nullable=False),
    StructField("customer_id",  StringType(), nullable=False),
    StructField("product_id",   StringType(), nullable=False),
    StructField("quantity",     IntegerType()),
    StructField("unit_price",   DoubleType()),
    StructField("discount_pct", DoubleType()),
    StructField("order_status", StringType()),
    StructField("channel",      StringType()),
    StructField("order_date",   DateType()),
    StructField("created_at",   TimestampType()),
])

raw_path = (f"s3://{BUCKET}/raw/orders/"
            f"year={year}/month={month}/day={day}/")

df_raw = (spark.read
    .option("header", True)
    .schema(orders_schema)
    .csv(raw_path))

logger.info(f"Read {df_raw.count()} rows from {raw_path}")

## ── 2. DATA QUALITY ────────────────────────────────
DataQualityChecker(df_raw, "raw_orders") \
    .check_nulls(["order_id", "customer_id", "product_id"]) \
    .check_duplicates("order_id") \
    .check_range("quantity", 1, 1000) \
    .check_range("unit_price", 0.01, 99999) \
    .summary()

## ── 3. TRANSFORM ───────────────────────────────────
VALID_STATUSES = ["PLACED", "SHIPPED", "DELIVERED", "RETURNED", "CANCELLED"]

df_clean = (df_raw
    # Standardise strings
    .withColumn("order_status", F.upper(F.trim(F.col("order_status"))))
    .withColumn("channel",      F.lower(F.trim(F.col("channel"))))
    # Filter invalid statuses
    .filter(F.col("order_status").isin(VALID_STATUSES))
    # Compute revenue metrics
    .withColumn("gross_revenue",
        F.round(F.col("quantity") * F.col("unit_price"), 2))
    .withColumn("net_revenue",
        F.round(F.col("gross_revenue") * (1 - F.col("discount_pct") / 100), 2))
    # Add audit columns
    .withColumn("processed_at", F.current_timestamp())
    .withColumn("process_date",  F.lit(PROC_DT))
    # Partition columns
    .withColumn("year",  F.lit(year))
    .withColumn("month", F.lit(month))
    .withColumn("day",   F.lit(day))
)

## ── 4. WRITE CURATED (Parquet, partitioned) ────────
curated_path = f"s3://{BUCKET}/curated/fact_orders/"

(df_clean.write
    .mode("overwrite")
    .partitionBy("year", "month", "day")
    .parquet(curated_path))

logger.info(f"✅ Wrote curated data to {curated_path}")

Performance Optimizations

  • Use schema enforcement — avoids schema inference scan on every run
  • Partition by date — eliminates full scans on downstream Athena queries
  • partitionBy("year","month","day") on write creates Hive-compatible partitions
  • Use broadcast() for small dimension tables to avoid shuffle joins
  • Parquet format — columnar, compressed, 5-10x smaller than CSV
