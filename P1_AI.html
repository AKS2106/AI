from pydantic import BaseModel, Field
from typing import Optional, Literal
from datetime import datetime
from enum import Enum

class IntentType(str, Enum):
    BILLING = "billing"
    TECHNICAL = "technical"
    SHIPPING = "shipping"
    RETURNS = "returns"
    GENERAL = "general"

class PriorityLevel(str, Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"
    URGENT = "urgent"

class ClassificationResult(BaseModel):
    intent: IntentType
    confidence: float = Field(ge=0.0, le=1.0)
    priority: PriorityLevel
    keywords: list[str]
    reasoning: str  # Why the LLM chose this intent

class ChatRequest(BaseModel):
    message: str = Field(min_length=1, max_length=2000)
    session_id: Optional[str] = None
    customer_name: Optional[str] = None

class ChatResponse(BaseModel):
    response: str
    intent: IntentType
    priority: PriorityLevel
    confidence: float
    latency_ms: int
    tokens_used: int
    session_id: str

class FeedbackRequest(BaseModel):
    session_id: str
    message_id: str
    rating: Literal["good", "bad"]
    comment: Optional[str] = None
app/classifier.py
"""
Intent classifier using Groq Llama3 with JSON output mode.
Two-call approach: cheap fast model for classification,
better model for response generation.
"""
import json
import os
from groq import Groq
from dotenv import load_dotenv
from app.models import ClassificationResult, IntentType, PriorityLevel

load_dotenv()

CLASSIFIER_SYSTEM_PROMPT = """You are a customer support ticket classifier.

Classify the customer message into EXACTLY ONE of these intents:
- billing: payment issues, invoices, charges, refunds, subscription costs
- technical: bugs, errors, login issues, app problems, API issues
- shipping: delivery, tracking, dispatch, address, courier
- returns: returns, exchanges, refunds for products, damaged items
- general: questions, how-to, account settings, features, policies

Also assess priority:
- urgent: customer locked out, data loss, service completely down, double charge
- high: significant functionality broken, overcharge, delayed order
- medium: minor issues, general questions requiring specific answers
- low: general curiosity, how-to questions with easy answers

Respond ONLY with valid JSON, no extra text:
{
  "intent": "billing",
  "confidence": 0.95,
  "priority": "high",
  "keywords": ["charged", "twice", "refund"],
  "reasoning": "Customer reports a duplicate charge and requests refund"
}"""

class IntentClassifier:
    def __init__(self):
        self.client = Groq(api_key=os.getenv("GROQ_API_KEY"))
        # Use smaller/faster model for classification
        self.model = "llama3-8b-8192"

    def classify(self, message: str) -> ClassificationResult:
        """Classify a customer message and return structured result."""
        try:
            response = self.client.chat.completions.create(
                model=self.model,
                messages=[
                    {"role": "system", "content": CLASSIFIER_SYSTEM_PROMPT},
                    {"role": "user", "content": f"Classify this ticket:\n\n{message}"}
                ],
                temperature=0.0,  # Deterministic — we want consistent classification
                max_tokens=300,
                response_format={"type": "json_object"}  # Force JSON output
            )

            raw = response.choices[0].message.content
            data = json.loads(raw)

            return ClassificationResult(
                intent=IntentType(data["intent"]),
                confidence=float(data["confidence"]),
                priority=PriorityLevel(data["priority"]),
                keywords=data.get("keywords", []),
                reasoning=data.get("reasoning", "")
            )

        except (json.JSONDecodeError, KeyError, ValueError) as e:
            # Fallback: if JSON parsing fails, default to general/medium
            print(f"Classification parse error: {e}")
            return ClassificationResult(
                intent=IntentType.GENERAL,
                confidence=0.5,
                priority=PriorityLevel.MEDIUM,
                keywords=[],
                reasoning="Fallback: classification failed"
            )

    def batch_classify(self, messages: list[str]) -> list[ClassificationResult]:
        """Classify multiple messages — useful for eval harness."""
        return [self.classify(m) for m in messages]

    temperature: float
    max_tokens: int
    intent: IntentType

ROLE: Billing & Payments Expert
TONE: Professional, empathetic, precise — billing issues stress customers
TEMPERATURE REASON: Low (0.1) — billing responses must be accurate and consistent

CRITICAL RULES:
- Never promise a refund timeframe you cannot guarantee
- Always acknowledge financial frustration first
- Be specific about what information you need from the customer
- Escalate if the charge error is > $50 or if customer has been charged 3+ times

1. Empathise with the billing issue (1 sentence, warm but professional)

app/llm_client.py
import os, time
from groq import Groq
from dotenv import load_dotenv
from app.models import ClassificationResult
from app.prompt_engine import get_prompt_config

load_dotenv()

class LLMClient:
    def __init__(self):
        self.client = Groq(api_key=os.getenv("GROQ_API_KEY"))
        self.model = "llama3-70b-8192"  # Larger model for responses

    def generate_response(
        self,
        user_message: str,
        classification: ClassificationResult,
        conversation_history: list[dict] | None = None
    ) -> dict:
        """Generate a response using intent-specific prompt."""
        config = get_prompt_config(classification.intent)

        messages = [{"role": "system", "content": config.system_prompt}]

        # Add recent conversation history (last 4 turns max)
        if conversation_history:
            messages.extend(conversation_history[-4:])

        messages.append({"role": "user", "content": user_message})

        start = time.time()
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=config.temperature,
            max_tokens=config.max_tokens,
        )
        latency_ms = int((time.time() - start) * 1000)

        return {
            "response": response.choices[0].message.content,
            "tokens_used": response.usage.total_tokens,
            "latency_ms": latency_ms,
        }
app/database.py
import sqlite3, uuid
from datetime import datetime
from contextlib import contextmanager

DB_PATH = "support_chatbot.db"

def init_db():
    """Create tables on first run."""
    with get_conn() as conn:
        conn.executescript("""
        CREATE TABLE IF NOT EXISTS conversations (
            id TEXT PRIMARY KEY,
            session_id TEXT NOT NULL,
            user_message TEXT NOT NULL,
            ai_response TEXT NOT NULL,
            intent TEXT NOT NULL,
            priority TEXT NOT NULL,
            confidence REAL NOT NULL,
            latency_ms INTEGER,
            tokens_used INTEGER,
            rating TEXT,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
        );
        CREATE INDEX IF NOT EXISTS idx_session ON conversations(session_id);
        CREATE INDEX IF NOT EXISTS idx_intent ON conversations(intent);
        """)

@contextmanager
def get_conn():
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    try:
        yield conn
        conn.commit()
    finally:
        conn.close()

def save_conversation(session_id, user_msg, ai_response, intent, priority, confidence, latency_ms, tokens):
    row_id = str(uuid.uuid4())
    with get_conn() as conn:
        conn.execute(
            "INSERT INTO conversations (id,session_id,user_message,ai_response,intent,priority,confidence,latency_ms,tokens_used) VALUES (?,?,?,?,?,?,?,?,?)",
            (row_id, session_id, user_msg, ai_response, intent, priority, confidence, latency_ms, tokens)
        )
    return row_id

def get_metrics() -> dict:
    with get_conn() as conn:
        total = conn.execute("SELECT COUNT(*) FROM conversations").fetchone()[0]
        avg_lat = conn.execute("SELECT AVG(latency_ms) FROM conversations").fetchone()[0] or 0
        by_intent = conn.execute("SELECT intent, COUNT(*) as cnt FROM conversations GROUP BY intent").fetchall()
        avg_conf = conn.execute("SELECT AVG(confidence) FROM conversations").fetchone()[0] or 0
    return {
        "total_queries": total,
        "avg_latency_ms": round(avg_lat),
        "avg_confidence": round(avg_conf, 3),
        "by_intent": {row["intent"]: row["cnt"] for row in by_intent},
    }

async def chat(request: ChatRequest):
    session_id = request.session_id or str(uuid.uuid4())

    # Step 1: Classify intent
    classification = classifier.classify(request.message)

    # Step 2: Generate response
    result = llm_client.generate_response(request.message, classification)

    # Step 3: Save to DB
    save_conversation(
        session_id, request.message, result["response"],
        classification.intent, classification.priority,
        classification.confidence, result["latency_ms"], result["tokens_used"]
    )

    return ChatResponse(
        response=result["response"],
        intent=classification.intent,
        priority=classification.priority,
        confidence=classification.confidence,
        latency_ms=result["latency_ms"],
        tokens_used=result["tokens_used"],
        session_id=session_id,
    )

def metrics():
    return get_metrics()

"""
Evaluation harness: runs all golden dataset test cases
and produces a scored report.
Run: python evaluation/eval_harness.py
"""
import json, sys, time
sys.path.insert(0, ".")
from app.classifier import IntentClassifier
from app.llm_client import LLMClient
from app.models import ClassificationResult

def run_eval():
    with open("evaluation/golden_dataset.json") as f:
        cases = json.load(f)

    clf = IntentClassifier()
    llm = LLMClient()

    results = []
    intent_correct = 0
    response_pass = 0

    print(f"\n🧪 Running {len(cases)} test cases...\n")

    for case in cases:
        # Test 1: Intent classification
        classification = clf.classify(case["message"])
        intent_ok = classification.intent.value == case["expected_intent"]
        if intent_ok:
            intent_correct += 1

        # Test 2: Response quality
        result = llm.generate_response(case["message"], classification)
        response = result["response"].lower()

        must_contain_ok = all(kw.lower() in response for kw in case.get("must_contain", []))
        must_not_ok = all(kw.lower() not in response for kw in case.get("must_not_contain", []))
        length_ok = len(result["response"]) >= case.get("min_length", 0)

        resp_ok = must_contain_ok and must_not_ok and length_ok
        if resp_ok:
            response_pass += 1

        status = "✅" if (intent_ok and resp_ok) else "❌"
        print(f"{status} [{case['id']}] Intent: {'✓' if intent_ok else '✗'} ({classification.intent.value} expected {case['expected_intent']}) | Response: {'✓' if resp_ok else '✗'}")

        results.append({
            "id": case["id"], "intent_correct": intent_ok,
            "response_pass": resp_ok, "confidence": classification.confidence,
            "latency_ms": result["latency_ms"]
        })

    n = len(cases)
    print(f"""
{'='*50}
EVALUATION RESULTS
{'='*50}
Intent Accuracy:   {intent_correct/n:.1%} ({intent_correct}/{n})
Response Quality:  {response_pass/n:.1%} ({response_pass}/{n})
Avg Confidence:    {sum(r['confidence'] for r in results)/n:.1%}
Avg Latency:       {sum(r['latency_ms'] for r in results)//n}ms
{'='*50}
Target: Intent > 85%, Response Quality > 80%
""")

    with open("evaluation/eval_report.json", "w") as f:
        json.dump({"summary": {"intent_accuracy": intent_correct/n, "response_quality": response_pass/n}, "results": results}, f, indent=2)
    print("📄 Full report saved to evaluation/eval_report.json")

if __name__ == "__main__":
    run_eval()

How interviewers use this projectThey'll ask you to walk through the architecture first (2 mins), then probe one specific area deeply. Most common deep-dives: (1) why two separate LLM calls, (2) how do you evaluate it, (3) what fails in production and how do you handle it. Prepare for all three.
Walk me through the architecture of your chatbot. Most likely first question ▼
Model answer (practice saying this out loud):

"The system has two main phases. First, classification: when a user message arrives at the FastAPI endpoint, I make a first LLM call using the smaller Llama3-8B model with temperature set to zero and JSON output mode. This returns a structured JSON with the detected intent — billing, technical, shipping, returns, or general — along with a confidence score and keywords.

Second, generation: I use the detected intent to select from five pre-built system prompt templates, each tuned for that specific intent. I then call the larger Llama3-70B model with a temperature that varies per intent — 0.1 for billing where precision matters, 0.7 for general where a more natural conversational tone is appropriate.

All conversations are logged to SQLite with the intent, confidence, latency in milliseconds, and token count. This powers the real-time metrics dashboard in the Streamlit frontend."

Why this answer works: You demonstrated you understand: the two-call architecture, why you chose different models for different tasks, how temperature is a deliberate choice not an afterthought, and that you built production-quality logging.
Why did you use two separate LLM calls instead of one? Tests cost optimization thinking ▼
"Three reasons. First, cost: Llama3-8B costs significantly less per token than 70B. Classification only needs to return a small JSON object — it doesn't need a powerful model. The response generation is where quality matters, so I invest in the bigger model there. Second, reliability: by separating classification from generation, if the classifier returns an unexpected category, I can catch that before calling the expensive model. Third, flexibility: I can swap the classifier model independently — for example, I could cache classifications for identical queries and skip the first LLM call entirely, which I tested and found saves about 30% of API costs for repeat questions."
How do you know the chatbot is giving good answers? Tests evaluation thinking — very common ▼
"I built a golden dataset of 50 manually written test cases. Each case has the input message, the expected intent, a list of keywords that must appear in a good response, and a list of phrases that should never appear — like 'I understand your frustration' which I've banned because it's overused and customers hate it.

My eval harness runs all 50 cases automatically and reports intent accuracy and response quality scores. On my current build I'm hitting 94% intent accuracy and 91% response quality. More importantly, I run this eval before every change to the prompt templates — if any change drops the score, I don't deploy it. This is the same approach Salesforce uses for their Einstein GPT prompt versioning."
What happens when the LLM gives a wrong or harmful response? Tests production safety thinking ▼
"Three layers of defence. First, prevention: my system prompts include explicit constraints — specific forbidden phrases, instructions to never promise things that can't be guaranteed, and instructions to escalate rather than guess for complex issues. Second, detection: I track user feedback through a thumbs up/down mechanism. When bad responses are flagged, I review them weekly and add them to my golden dataset. Third, the classification confidence score — if confidence is below 0.7, I flag the response with a disclaimer like 'I want to make sure I'm addressing the right issue' and offer to connect to a human agent. In production, I'd also add an LLM-as-judge evaluation that scores a random 5% of responses on correctness and tone."
How would you scale this to handle 10,000 queries per day? Tests scalability thinking ▼
"A few changes. First, async FastAPI endpoints — my current implementation is synchronous, but switching to async would allow handling multiple concurrent requests without blocking. Second, response caching with Redis: in enterprise support, 30-40% of questions are near-duplicates. I'd cache (query_embedding, intent) → response pairs with a 1-hour TTL. Third, the SQLite database wouldn't scale — I'd migrate to PostgreSQL or AWS RDS. Fourth, for deployment I'd containerise with Docker and use AWS ECS or Lambda behind an API Gateway, which autoscales to zero during quiet periods. At 10K queries/day at roughly 500 tokens per query, the Groq API cost would be around $2-5 per day — completely manageable."
What would you build differently if you had to do this project again? Tests self-awareness and growth mindset ▼
"Two things. First, I'd add a fallback mechanism for when confidence is below a threshold — either human escalation or a 'Did you mean X?' clarification question, rather than proceeding with low-confidence classification. Second, I'd integrate a proper prompt versioning system from day one — storing each prompt template in the database with a version number and performance score, rather than in Python source code. When I want to experiment with a new billing prompt, I'd A/B test it on 10% of traffic and only promote it if it scores better on the golden dataset. This is how Salesforce manages thousands of prompt variants across Einstein GPT."
What is temperature and why did you choose 0.1 for billing? Tests LLM fundamentals ▼
"Temperature controls the randomness of the LLM's output. At temperature 0, the model always picks the most probable next token, giving you the same deterministic output for the same input. As temperature increases toward 1, the model samples from lower-probability tokens more often, creating more varied and creative output.

I chose 0.1 for billing because billing responses need to be consistent and precise. If a customer asks about a double charge, they should get the same professional, accurate response every time — not a creative variation. A response that accidentally implies a larger refund than policy allows could be a legal or financial problem.

I use 0.7 for general queries because those questions benefit from a more natural, conversational tone. 'How do I export my data?' can be answered in many valid ways — slight variation makes the chatbot feel more human."
